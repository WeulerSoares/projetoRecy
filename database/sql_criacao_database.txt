CREATE DATABASE IF NOT EXISTS app_reciclagem;

USE app_reciclagem;

CREATE TABLE IF NOT EXISTS usuarios (
    id INT AUTO_INCREMENT PRIMARY KEY,
    firebase_uid VARCHAR(128) NOT NULL,
    nome VARCHAR(255),
    email VARCHAR(255) UNIQUE,
    data_nascimento DATE,
    pontos_acumulados INT DEFAULT 0 NOT NULL,
    data_registro DATE,
    tipo_usuario INT DEFAULT 0 NOT NULL,
    cpf CHAR(11) UNIQUE,
    cnpj CHAR(14) UNIQUE,
    foto_path VARCHAR(255),
    
    CONSTRAINT UC_email UNIQUE (email),
    INDEX idx_firebase_uid (firebase_uid),
    INDEX idx_tipo_usuario (tipo_usuario)
);

CREATE TABLE IF NOT EXISTS empresas_parceira (
    id_empresa INT AUTO_INCREMENT PRIMARY KEY,
    cnpj CHAR(14) UNIQUE,
    nome VARCHAR(255),
    email VARCHAR(255) UNIQUE,
    data_registro DATE,
    logo_path VARCHAR(255),
    
    CONSTRAINT UC_email UNIQUE (email)
);

CREATE TABLE cupons (
    id_cupom INT AUTO_INCREMENT PRIMARY KEY,
    id_empresa INT,
    valor DECIMAL(10, 2),
    quantidade INT,
    quantidade_estoque INT,
    pontos INT,
    FOREIGN KEY (id_empresa) REFERENCES empresas_parceira(id_empresa)
);

CREATE TABLE historico_resgate_cupons (
    id_resgate INT AUTO_INCREMENT PRIMARY KEY,
    id_cupom INT,
    id_usuario INT,
    data_resgate DATE,
    FOREIGN KEY (id_cupom) REFERENCES cupons(id_cupom),
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id)
);
